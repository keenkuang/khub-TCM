### 线 B 任务：运维可观测性

- [ ] **B1：重写 `khub/log.py`**

```python
"""结构化日志。"""
import json, logging, os, sys
from logging.handlers import TimedRotatingFileHandler

def setup_logging():
    level = os.environ.get("KHUB_LOG_LEVEL", "INFO").upper()
    fmt = os.environ.get("KHUB_LOG_FORMAT", "json")
    target = os.environ.get("KHUB_LOG_FILE", "")
    root = logging.getLogger()
    root.setLevel(level)
    handler = (logging.FileHandler(target) if target
               else logging.StreamHandler(sys.stderr))
    if fmt == "json":
        handler.setFormatter(JsonFormatter())
    else:
        handler.setFormatter(logging.Formatter(
            "%(asctime)s [%(levelname)s] %(name)s: %(message)s"))
    if target and fmt == "json":
        handler = TimedRotatingFileHandler(target, when="midnight",
                                           backupCount=int(os.environ.get("KHUB_LOG_ROTATION", "30")))
        handler.setFormatter(JsonFormatter())
    root.addHandler(handler)

class JsonFormatter(logging.Formatter):
    def format(self, record):
        return json.dumps({
            "timestamp": self.formatTime(record),
            "level": record.levelname,
            "module": record.name,
            "message": record.getMessage(),
        }, ensure_ascii=False)
```

- [ ] **B3：`/health` 深度增强**

`api.py` health 分支改为逐项检查：
```python
def _health_check(store):
    checks = {"db": {"ok": True, "documents": 0}, "disk": {"ok": True}}
    try:
        c = store.conn.execute("SELECT count(*) FROM documents").fetchone()[0]
        checks["db"] = {"ok": True, "documents": c}
    except Exception as e:
        checks["db"] = {"ok": False, "error": str(e)}
    try:
        import os
        st = os.statvfs(os.path.expanduser("~/.khub"))
        free_mb = st.f_bavail * st.f_frsize / 1024 / 1024
        checks["disk"] = {"ok": free_mb > 100, "free_mb": round(free_mb, 1)}
    except Exception:
        checks["disk"] = {"ok": True, "note": "unavailable"}
    all_ok = all(v.get("ok", True) for v in checks.values())
    return ("ok" if all_ok else "degraded"), checks
```
