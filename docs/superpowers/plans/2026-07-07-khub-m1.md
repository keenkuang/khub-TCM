**目标：** 搭出 khub 可运行的本地内核——SQLite+FTS5 存储（多版本）、适配器接口、Obsidian 适配器接口占位、中医 OCR 入库适配器、CLI（sync/query/ingest/ls/conflicts/version）、Sync 引擎与冲突多版本，并用隔离测试验证。不依赖 Obsidian 已安装。

**架构：** 纯 Python（标准库 + sqlite3/FTS5），无重依赖。内核 = `db`(schema) + `models`(数据类) + `adapters`(每源一个，统一接口) + `sync`(源无关引擎) + `query`(全文检索) + `cli`(入口)。所有界面/外部系统只通过适配器与内核交互。

**技术栈：** Python 3.11、sqlite3 + FTS5（trigram 分词以支持中文子串检索）、argparse（CLI）、pytest（测试）。

```python
import sqlite3
from pathlib import Path

SCHEMA_VERSION = 1

_SCHEMA = """
CREATE TABLE IF NOT EXISTS documents (
    id TEXT PRIMARY KEY,
    title TEXT NOT NULL,
    current_version TEXT,
    source_ids TEXT DEFAULT '[]',
    created_at TEXT DEFAULT (datetime('now')),
    updated_at TEXT DEFAULT (datetime('now')),
    conflict INTEGER NOT NULL DEFAULT 0
);
CREATE TABLE IF NOT EXISTS document_versions (
    version_id TEXT PRIMARY KEY,
    doc_id TEXT NOT NULL,
    content TEXT,
    format TEXT DEFAULT 'markdown',
    origin TEXT NOT NULL,
    author TEXT,
    updated_at TEXT,
    hash TEXT,
    parent_version TEXT,
    note TEXT
);
CREATE TABLE IF NOT EXISTS attachments (
    id TEXT PRIMARY KEY,
    doc_id TEXT,
    version_id TEXT,
    kind TEXT,
    path TEXT,
    hash TEXT
);
CREATE TABLE IF NOT EXISTS sync_states (
    source_id TEXT,
    doc_id TEXT,
    last_sync_at TEXT,
    etag TEXT,
    hash TEXT,
    PRIMARY KEY (source_id, doc_id)
);
CREATE TABLE IF NOT EXISTS embeddings (
    doc_id TEXT,
    version_id TEXT,
    model TEXT,
    vector BLOB
);
CREATE VIRTUAL TABLE IF NOT EXISTS doc_fts USING fts5(
    title, body, doc_id UNINDEXED, tokenize='trigram'
);
"""


def open_db(path: str) -> sqlite3.Connection:
    Path(path).parent.mkdir(parents=True, exist_ok=True)
    conn = sqlite3.connect(path)
    conn.execute("PRAGMA foreign_keys = ON")
    conn.executescript(_SCHEMA)
    conn.commit()
    return conn
```

`khub/adapters/base.py`：
```python
from abc import ABC, abstractmethod
from typing import List
from khub.models import RawDoc, CanonicalDoc


class SourceAdapter(ABC):
    # 子类设：name, direction("two-way"|"read-only")
    direction = "read-only"

    @abstractmethod
    def pull(self) -> List[RawDoc]:
        """列出并拉取源端文档（正文+附件+元数据）。"""

    @abstractmethod
    def normalize(self, raw: RawDoc) -> CanonicalDoc:
        """把源原始文档转为规范模型。"""

    def detect_changes(self, raw: RawDoc, last_hash: str | None) -> bool:
        """默认按内容哈希判断；子类可覆盖为 etag/updated_at。"""
        from khub.hashutil import sha256_text
        return sha256_text(raw.content or "") != last_hash

    def push(self, doc: CanonicalDoc) -> None:
        """写回源（read-only 源不实现）。"""
        raise NotImplementedError(f"{self.__class__.__name__} 不支持 push")
```

`khub/adapters/ocr.py`：
```python
from pathlib import Path
from typing import List
from khub.adapters.base import SourceAdapter
from khub.models import RawDoc


class OcrAdapter(SourceAdapter):
    direction = "read-only"  # 只入不出（生产者源）

    def __init__(self, name: str, inbox: str):
        self.name = name
        self.inbox = Path(inbox)

    def pull(self) -> List[RawDoc]:
        """扫描 inbox 下每本子目录，逐本 ingest。"""
        out: List[RawDoc] = []
        if not self.inbox.exists():
            return out
        for book in sorted(self.inbox.iterdir()):
            if book.is_dir():
                out.extend(self.ingest_book(str(book)))
        return out

    def ingest_book(self, book_dir: str) -> List[RawDoc]:
        book = Path(book_dir)
        index = book / "index.md"
        if not index.exists():
            # 退回：把目录下所有 .md 各当一篇
            raws = []
            for md in sorted(book.glob("*.md")):
                text = md.read_text(encoding="utf-8")
                raws.append(self._to_raw(md.stem, text, str(md)))
            return raws
        text = index.read_text(encoding="utf-8")
        return [self._to_raw(book.name, text, str(index))]

    def _to_raw(self, title: str, content: str, source_id: str) -> RawDoc:
        return RawDoc(source_id=source_id, title=title, content=content)

    def normalize(self, raw: RawDoc) -> "CanonicalDoc":  # noqa: F821
        from khub.models import CanonicalDoc
        return CanonicalDoc(
            id=f"ocr:{raw.source_id}",
            title=raw.title, content=raw.content,
            source="ocr", source_id=raw.source_id,
        )
```

### 任务 5：Sync 引擎（diff / pull / 冲突多版本）

**文件：**
- 创建：`khub/sync.py`
- 创建：`tests/test_sync.py`

核心：对每个源，pull → normalize → 与 `documents`/`sync_states` 比对：
- 全新 → 建文档 + 版本（origin=source）。
- 仅源变更 → 新版本（origin=source），更新 sync_states。
- `two-way` 且本地自上次同步后有改动（document.updated_at > last_sync_at 且 hash 不同）→ push 回源，更新 sync_states。
- **冲突（源变且本地也变且内容不同）** → 不覆盖：新增两个版本（origin=source 与 origin=hub），`documents.conflict=1`，sync_states 记录两端 hash。

`khub/sync.py`：
```python
import sqlite3
from datetime import datetime
from khub.models import CanonicalDoc


def _now() -> str:
    return datetime.now().isoformat(timespec="seconds")


def _new_version_id(doc_id: str, origin: str) -> str:
    return f"{doc_id}:{origin}:{_now()}"


def _upsert_doc(conn, doc: CanonicalDoc, origin: str, content: str, etag=None, hash=None):
    cur = conn.cursor()
    existing = cur.execute("SELECT id FROM documents WHERE id=?", (doc.id,)).fetchone()
    vid = _new_version_id(doc.id, origin)
    cur.execute(
        """INSERT INTO document_versions(version_id,doc_id,content,origin,author,updated_at,hash)
           VALUES(?,?,?,?,?,?,?)""",
        (vid, doc.id, content, origin, doc.author, _now(), hash))
    if existing is None:
        cur.execute(
            "INSERT INTO documents(id,title,current_version,source_ids,updated_at) VALUES(?,?,?,?,?)",
            (doc.id, doc.title, vid, '["%s"]' % doc.source, _now()))
    else:
        cur.execute("UPDATE documents SET current_version=?,updated_at=? WHERE id=?",
                    (vid, _now(), doc.id))
    cur.execute(
        """INSERT INTO sync_states(source_id,doc_id,last_sync_at,etag,hash)
           VALUES(?,?,?,?,?) ON CONFLICT(source_id,doc_id)
           DO UPDATE SET last_sync_at=excluded.last_sync_at,etag=excluded.etag,hash=excluded.hash""",
        (doc.source, doc.id, _now(), etag, hash))
    # 同步 FTS 索引（用最新版本内容）
    cur.execute("DELETE FROM doc_fts WHERE doc_id=?", (doc.id,))
    cur.execute("INSERT INTO doc_fts(doc_id,title,body) VALUES(?,?,?)",
                (doc.id, doc.title, content))
    conn.commit()
    return vid


def local_edit(conn, doc_id: str, new_content: str):
    """在 hub 侧修改内容（制造本地版本，供冲突测试/真实编辑）。"""
    _upsert_doc(conn, CanonicalDoc(id=doc_id, title="", content=new_content, source="hub"),
                "hub", new_content, hash="local")


def sync_source(conn: sqlite3.Connection, adapter) -> dict:
    raws = adapter.pull()
    stats = {"created": 0, "updated": 0, "conflicts": 0, "pushed": 0}
    for raw in raws:
        doc = adapter.normalize(raw)
        last = conn.execute(
            "SELECT hash FROM sync_states WHERE source_id=? AND doc_id=?",
            (doc.source, doc.id)).fetchone()
        last_hash = last[0] if last else None
        changed = adapter.detect_changes(raw, last_hash)

        hub_changed = conn.execute(
            "SELECT 1 FROM document_versions WHERE doc_id=? AND origin='hub' "
            "AND hash<>?", (doc.id, _hash(raw.content))).fetchone() if False else None
        # 简化：以 sync_states 之外是否存在 hub 版本且内容不同于源来判断本地改动
        hub_ver = conn.execute(
            "SELECT hash FROM document_versions WHERE doc_id=? AND origin='hub' ORDER BY rowid DESC LIMIT 1",
            (doc.id,)).fetchone()
        local_dirty = hub_ver is not None and hub_ver[0] != _hash(raw.content)

        if not last:  # 全新
            _upsert_doc(conn, doc, doc.source, raw.content, raw.etag, _hash(raw.content))
            stats["created"] += 1
        elif changed and local_dirty and adapter.direction == "two-way":
            # 冲突：源变 + 本地变 → 两个版本都不覆盖
            _upsert_doc(conn, doc, doc.source, raw.content, raw.etag, _hash(raw.content))
            # hub 版本已在 local_edit 时写入；标记冲突
            conn.execute("UPDATE documents SET conflict=1 WHERE id=?", (doc.id,))
            stats["conflicts"] += 1
        elif changed:
            _upsert_doc(conn, doc, doc.source, raw.content, raw.etag, _hash(raw.content))
            stats["updated"] += 1
        else:
            continue

        # two-way：若本地有未推送改动且源无冲突，推回源
        if adapter.direction == "two-way" and local_dirty and not stats["conflicts"]:
            adapter.push(doc)
            stats["pushed"] += 1
    return stats


def _hash(text: str) -> str:
    from khub.hashutil import sha256_text
    return sha256_text(text or "")
```

> 注：`sync_source` 中本地改动判定用 `document_versions(origin='hub')` 的最新 hash 与源内容比对；真实编辑走 `local_edit()`。冲突分支确保不覆盖、双版本并存。

```python
import sqlite3


def search(conn: sqlite3.Connection, query: str, limit: int = 20) -> list:
    rows = conn.execute(
        """SELECT d.doc_id, d.title, snippet(doc_fts, 2, '[', ']', '…', 10)
```

```python
import argparse
import sqlite3
import sys
from khub.db import open_db
from khub.sync import sync_source
from khub.query import search
from khub.adapters import OcrAdapter, ObsidianAdapter


def build_parser() -> argparse.ArgumentParser:
    p = argparse.ArgumentParser(prog="khub", description="个人知识库中枢")
    sub = p.add_subparsers(dest="cmd", required=True)

    sp = sub.add_parser("sync", help="同步所有/指定源")
    sp.add_argument("--source", default=None)

    qp = sub.add_parser("query", help="全文检索")
    qp.add_argument("text")

    ip = sub.add_parser("ingest", help="OCR 入库一本")
    ip.add_argument("--book", required=True)

    sub.add_parser("ls", help="列出文档")
    sub.add_parser("conflicts", help="列出冲突")
    sub.add_parser("version", help="查看版本历史")
    return p


def main(argv=None):
    args = build_parser().parse_args(argv)
    db_path = "khub.db"
    conn = open_db(db_path)
    if args.cmd == "sync":
        for ad in (OcrAdapter(name="ocr", inbox="inbox/ocr"),):
            print(sync_source(conn, ad))
    elif args.cmd == "query":
        for h in search(conn, args.text):
            print(f"[{h['doc_id']}] {h['title']}\n  {h['snippet']}")
    elif args.cmd == "ingest":
        ad = OcrAdapter(name="ocr", inbox="inbox/ocr")
        print(ad.ingest_book(args.book))
    elif args.cmd == "ls":
        for r in conn.execute("SELECT id,title,conflict FROM documents"):
            print(r)
    elif args.cmd == "conflicts":
        for r in conn.execute("SELECT id,current_version FROM documents WHERE conflict=1"):
            print(r)
    elif args.cmd == "version":
        print("use: khub ls / 直接查 document_versions")
    conn.close()


if __name__ == "__main__":
    main()
```
