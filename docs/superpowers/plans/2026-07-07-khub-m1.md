```python
import sqlite3
import hashlib
import json
import time
from typing import Optional
from .models import CanonicalDoc


def compute_hash(text: str) -> str:
    return hashlib.sha256(text.encode("utf-8")).hexdigest()


def _now() -> str:
    return time.strftime("%Y-%m-%dT%H:%M:%S.%fZ", time.gmtime())


MAX_DOC_SIZE = 10 * 1024 * 1024  # 10 MB


class Store:
    def __init__(self, path=":memory:"):
        self.path = path
        self.conn = sqlite3.connect(path)
        self.conn.row_factory = sqlite3.Row
        self.conn.execute("PRAGMA journal_mode=WAL")
        self.conn.execute("PRAGMA busy_timeout=5000")
        self._ensure_fts5()
        self.init_schema()
        if path != ":memory:":
            import os
            os.chmod(path, 0o600)

    def _ensure_fts5(self):
        try:
            self.conn.execute("CREATE VIRTUAL TABLE _khub_fts_test USING fts5(x)")
            self.conn.execute("DROP TABLE _khub_fts_test")
        except sqlite3.OperationalError:
            raise RuntimeError(
                "SQLite FTS5 不可用。khub 需要 FTS5 支持。\n"
                "Ubuntu/Debian: sudo apt install libsqlite3-mod-fts5\n"
                "其他系统: 请确认 Python sqlite3 编译时启用了 FTS5")

    def init_schema(self):
        cur = self.conn
        cur.executescript("""
        CREATE TABLE IF NOT EXISTS sources(
            name TEXT PRIMARY KEY, type TEXT, direction TEXT, config_ref TEXT);
        CREATE TABLE IF NOT EXISTS documents(
            canonical_id TEXT PRIMARY KEY, title TEXT, current_version INTEGER,
            source_ids TEXT, created_at TEXT, updated_at TEXT,
            doc_type TEXT DEFAULT 'raw', conflict INTEGER DEFAULT 0);
        CREATE TABLE IF NOT EXISTS document_versions(
            version_id INTEGER PRIMARY KEY AUTOINCREMENT, doc_id TEXT,
            content TEXT, format TEXT, origin TEXT, author TEXT,
            updated_at TEXT, hash TEXT, parent_version INTEGER, note TEXT);
        CREATE TABLE IF NOT EXISTS attachments(
            id INTEGER PRIMARY KEY AUTOINCREMENT, doc_id TEXT, version_id INTEGER,
            kind TEXT, path TEXT, hash TEXT);
        CREATE TABLE IF NOT EXISTS sync_states(
            source_id TEXT, doc_id TEXT, last_sync_at TEXT, etag TEXT, hash TEXT,
            PRIMARY KEY(source_id, doc_id));
        CREATE TABLE IF NOT EXISTS embeddings(
            doc_id TEXT, version_id INTEGER, model TEXT, vector BLOB);
        CREATE TABLE IF NOT EXISTS schema_meta(
            key TEXT PRIMARY KEY, value TEXT);
        CREATE VIRTUAL TABLE IF NOT EXISTS docs_fts USING fts5(
            doc_id, title, content, tokenize='trigram');
        INSERT INTO schema_meta(key, value) VALUES('schema_version', '1')
            ON CONFLICT(key) DO NOTHING;
        """)

    def store_document(self, doc: CanonicalDoc, parent_version: Optional[int] = None) -> int:
        if len(doc.content) > MAX_DOC_SIZE:
            raise ValueError(f"文档内容超过上限 ({MAX_DOC_SIZE} bytes): {doc.canonical_id}")
        cur = self.conn
        existing = cur.execute(
            "SELECT canonical_id, current_version FROM documents WHERE canonical_id=?",
            (doc.canonical_id,)).fetchone()
        if existing is None:
            cur.execute(
                "INSERT INTO documents(canonical_id, title, current_version, source_ids, "
                "created_at, updated_at, doc_type, conflict) VALUES(?,?,?,?,?,?,?,0)",
                (doc.canonical_id, doc.title, 1, json.dumps([doc.source]),
                 _now(), doc.updated_at or _now(), doc.doc_type))
            parent = parent_version
        else:
            parent = parent_version if parent_version is not None else existing["current_version"]
        cur.execute(
            "INSERT INTO document_versions(doc_id, content, format, origin, author, "
            "updated_at, hash, parent_version, note) VALUES(?,?,?,?,?,?,?,?,?)",
            (doc.canonical_id, doc.content, doc.format, doc.origin, "",
             doc.updated_at or _now(), doc.hash or compute_hash(doc.content),
             parent, doc.note))
        version_id = cur.lastrowid
        cur.execute("UPDATE documents SET current_version=?, title=?, updated_at=? "
                    "WHERE canonical_id=?", (version_id, doc.title,
                     doc.updated_at or _now(), doc.canonical_id))
        cur.execute("DELETE FROM docs_fts WHERE doc_id=?", (doc.canonical_id,))
        cur.execute("INSERT INTO docs_fts(doc_id, title, content) VALUES(?,?,?)",
                    (doc.canonical_id, doc.title, doc.content))
        for a in doc.attachments:
            cur.execute("INSERT INTO attachments(doc_id, version_id, kind, path, hash) "
                        "VALUES(?,?,?,?,?)", (doc.canonical_id, version_id, a.kind, a.path, a.hash))
        self.conn.commit()
        return version_id

    def get_document(self, canonical_id: str):
        return self.conn.execute("SELECT * FROM documents WHERE canonical_id=?",
                                 (canonical_id,)).fetchone()

    def get_versions(self, doc_id: str):
        return self.conn.execute(
            "SELECT * FROM document_versions WHERE doc_id=? ORDER BY version_id",
            (doc_id,)).fetchall()

    def set_sync_state(self, source_id: str, doc_id: str, etag: str, h: str):
        self.conn.execute(
            "INSERT INTO sync_states(source_id, doc_id, last_sync_at, etag, hash) "
            "VALUES(?,?,?,?,?) ON CONFLICT(source_id, doc_id) DO UPDATE SET "
            "last_sync_at=excluded.last_sync_at, etag=excluded.etag, hash=excluded.hash",
            (source_id, doc_id, _now(), etag, h))
        self.conn.commit()

    def get_sync_state(self, source_id: str, doc_id: str):
        return self.conn.execute(
            "SELECT * FROM sync_states WHERE source_id=? AND doc_id=?",
            (source_id, doc_id)).fetchone()

    def mark_conflict(self, doc_id: str, flag: bool = True):
        self.conn.execute("UPDATE documents SET conflict=? WHERE canonical_id=?",
                          (1 if flag else 0, doc_id))
        self.conn.commit()

    def search(self, text: str):
        if len(text) > 200:
            text = text[:200]
        try:
            rows = self.conn.execute(
                "SELECT doc_id, title, snippet(docs_fts, 2, '[', ']', '...', 10) AS snip "
                "FROM docs_fts WHERE docs_fts MATCH ?", (text,)).fetchall()
        except sqlite3.DatabaseError:
            rows = []
        return [(r["doc_id"], r["title"], r["snip"]) for r in rows]

    def get_current_hash(self, doc_id: str) -> Optional[str]:
        row = self.conn.execute(
            "SELECT hash FROM document_versions WHERE doc_id=? ORDER BY version_id DESC LIMIT 1",
            (doc_id,)).fetchone()
        return row["hash"] if row else None

    def list_conflicts(self):
        rows = self.conn.execute("SELECT canonical_id FROM documents WHERE conflict=1").fetchall()
        return [r["canonical_id"] for r in rows]
```

```python
from abc import ABC, abstractmethod
from typing import Optional
from khub.models import RawDoc, CanonicalDoc
from khub.db import compute_hash


class SourceAdapter(ABC):
    name: str
    direction: str  # "two-way" | "read-only" | "push-in"

    @abstractmethod
    def pull(self) -> list[RawDoc]:
        ...

    @abstractmethod
    def push(self, doc: CanonicalDoc):
        ...

    @abstractmethod
    def normalize(self, raw: RawDoc) -> CanonicalDoc:
        ...

    def detect_changes(self, raw: RawDoc, last_state: Optional[object]) -> bool:
        if last_state is None:
            return True
        new_hash = raw.etag or compute_hash(raw.content)
        old = None
        if last_state is not None:
            state = dict(last_state)
            old = state.get("hash") or state.get("etag")
        if old is None:
            return True
        return new_hash != old
```

import hashlib
import os
from pathlib import Path
from khub.adapters.base import SourceAdapter
from khub.models import RawDoc, CanonicalDoc, Attachment


class OcrAdapter(SourceAdapter):
    name = "ocr"
    direction = "push-in"

    def __init__(self, book_dir: str):
        self.book_dir = Path(book_dir)
        if not self.book_dir.is_dir():
            raise ValueError(f"OCR book_dir 不存在或非目录：{book_dir}")

    def pull(self) -> list[RawDoc]:
        docs = []
        for md in sorted(self.book_dir.glob("*.md")):
            content = md.read_text(encoding="utf-8")
            etag = hashlib.sha256(content.encode("utf-8")).hexdigest()
            atts = []
            att_dir = self.book_dir / "attachments"
            if att_dir.is_dir():
                for f in sorted(att_dir.iterdir()):
                    if f.is_file():
                        try:
                            atts.append(Attachment(kind=f.suffix.lstrip(".") or "file",
                                                    path=str(f),
                                                    hash=hashlib.sha256(f.read_bytes()).hexdigest()))
                        except (IOError, PermissionError) as e:
                            import logging
                            logging.warning("跳过附件 %s: %s", f, e)
            docs.append(RawDoc(id=md.stem, title=md.stem, content=content,
                               etag=etag, attachments=atts))
        return docs

    def normalize(self, raw: RawDoc) -> CanonicalDoc:
        return CanonicalDoc(
            canonical_id=f"ocr/{raw.id}",
            title=raw.title, content=raw.content, source="ocr",
            source_id=f"ocr/{raw.id}", hash=raw.etag,
            attachments=raw.attachments)

    def push(self, doc: CanonicalDoc):
        raise NotImplementedError("ocr 为 push-in 源，不写回")
```

from khub.db import Store
from khub.models import CanonicalDoc, SyncResult


class SyncEngine:
    def __init__(self, store: Store):
        self.store = store

    def ingest(self, adapter) -> list[SyncResult]:
        results = []
        for raw in adapter.pull():
            try:
                canon = adapter.normalize(raw)
                canon.origin = "source"
                vid = self.store.store_document(canon)
                self.store.set_sync_state(adapter.name, canon.source_id, canon.hash, canon.hash)
                results.append(SyncResult("created", canon.canonical_id, vid))
            except Exception as e:
                import logging
                logging.error("[ingest] %s/%s 处理失败: %s", adapter.name, raw.id, e)
                results.append(SyncResult("error", raw.id, message=str(e)))
        return results

    def sync_source(self, adapter) -> list[SyncResult]:
        results = []
        for raw in adapter.pull():
            try:
                canon = adapter.normalize(raw)
                state = self.store.get_sync_state(adapter.name, canon.source_id)
                if state is None:
                    canon.origin = "source"
                    vid = self.store.store_document(canon)
                    self.store.set_sync_state(adapter.name, canon.source_id, canon.hash, canon.hash)
                    results.append(SyncResult("created", canon.canonical_id, vid))
                    continue
                if not adapter.detect_changes(raw, state):
                    continue
                canon.origin = "source"
                vid = self.store.store_document(canon)
                self.store.set_sync_state(adapter.name, canon.source_id, canon.hash, canon.hash)
                results.append(SyncResult("updated", canon.canonical_id, vid))
            except Exception as e:
                import logging
                logging.error("[sync_source] %s/%s 处理失败: %s", adapter.name, raw.id, e)
                results.append(SyncResult("error", raw.id, message=str(e)))
        return results

    def query(self, text: str):
        return self.store.search(text)

    def conflicts(self):
        return self.store.list_conflicts()

    def versions(self, doc_id: str):
        return self.store.get_versions(doc_id)
```

import os
import sys
import argparse
from khub.db import Store
from khub.engine import SyncEngine

from khub.config import load_config

CONFIG_PATH = os.environ.get("KHUB_CONFIG", "/home/keen/khub/config.yaml")

def _store():
    if not os.path.exists(CONFIG_PATH):
        if "KHUB_CONFIG" in os.environ:
            raise RuntimeError(f"指定的配置文件不存在：{CONFIG_PATH}")
        print(f"警告: 配置文件 {CONFIG_PATH} 不存在，使用默认配置", file=sys.stderr)
        return Store("/home/keen/khub/khub.db")
    try:
        cfg = load_config(CONFIG_PATH)
    except Exception as e:
        raise RuntimeError(f"加载配置失败 {CONFIG_PATH}: {e}")
    return Store(cfg.get("db", "/home/keen/khub/khub.db"))

def build_parser():
    p = argparse.ArgumentParser(prog="khub", description="个人知识库中枢")
    sub = p.add_subparsers(dest="cmd", required=True)
    sub.add_parser("sync").add_argument("--source", default=None)
    sub.add_parser("query").add_argument("text")
    ing = sub.add_parser("ingest"); ing.add_argument("--book", required=True)
    ls = sub.add_parser("ls"); ls.add_argument("--source", default=None)
    sub.add_parser("conflicts")
    sub.add_parser("version").add_argument("doc")
    return p

def main(argv=None):
    import logging
    logging.basicConfig(level=logging.INFO, format="%(levelname)s\t%(message)s")
    args = build_parser().parse_args(argv)
    store = _store()
    eng = SyncEngine(store)

    elif args.cmd == "conflicts":

    elif args.cmd == "version":
        for v in eng.versions(args.doc):
            print(f"v{v['version_id']}\t{v['origin']}\t{v['updated_at']}")
    return 0

if __name__ == "__main__":
    sys.exit(main())
