- [ ] **B2：创建 `khub/tags.py`**

```python
def add_tag(store, doc_id: str, tag: str):
    store.conn.execute("INSERT OR IGNORE INTO doc_tags (doc_id, tag) VALUES (?, ?)", (doc_id, tag))

def remove_tag(store, doc_id: str, tag: str):
    store.conn.execute("DELETE FROM doc_tags WHERE doc_id=? AND tag=?", (doc_id, tag))

def list_tags(store) -> list[dict]:
    return store.conn.execute("SELECT tag, count(*) as count FROM doc_tags GROUP BY tag ORDER BY count DESC").fetchall()

def get_doc_tags(store, doc_id: str) -> list[str]:
    rows = store.conn.execute("SELECT tag FROM doc_tags WHERE doc_id=?", (doc_id,)).fetchall()
    return [r["tag"] for r in rows]
```

- [ ] **B3：创建 `khub/favorites.py`**

```python
def toggle_favorite(store, doc_id: str) -> bool:
    row = store.conn.execute("SELECT id FROM favorites WHERE doc_id=?", (doc_id,)).fetchone()
    if row:
        store.conn.execute("DELETE FROM favorites WHERE doc_id=?", (doc_id,))
        return False  # 取消收藏
    else:
        store.conn.execute("INSERT INTO favorites (doc_id) VALUES (?)", (doc_id,))
        return True  # 收藏

def list_favorites(store) -> list[dict]:
    return store.conn.execute(
        "SELECT f.doc_id, d.title, f.created_at FROM favorites f LEFT JOIN documents d ON f.doc_id=d.canonical_id ORDER BY f.created_at DESC").fetchall()
```
